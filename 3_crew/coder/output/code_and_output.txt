```python
# Calculate the first 10,000 terms of the series
def calculate_series(terms):
    total_sum = 0
    for n in range(terms):
        # Calculate each term
        term = (-1) ** n / (2 * n + 1)
        total_sum += term
    # Multiply the result by 4
    return total_sum * 4

# Set the number of terms
num_terms = 10000
result = calculate_series(num_terms)
print(result)
```
